def self.break_sponsor_links(sponsor, sponsoree)
  links = TrustLink.where(:user_id => sponsor.id, :trustee => sponsoree.id)
  links.each do |link|
    link.destroy
  end
  links = TrustLink.where(:user_id => sponsoree.id, :trustee => sponsor.id)
  links.each do |link|
    link.destroy
  end 
end
